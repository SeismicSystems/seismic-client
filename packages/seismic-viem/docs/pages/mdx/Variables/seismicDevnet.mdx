# seismicDevnet

<div className="doc-side">
<div className="doc-main-source">

<div className="doc-subtitle">

Defines the Seismic development network configuration.

This configuration includes the chain ID, name, native currency, and RPC URLs
for connecting to the Seismic development network. It can be used in applications
that interact with the blockchain using a defined chain.

</div>

```ts
const seismicDevnet: {
  blockExplorers?: {
    default: ChainBlockExplorer;
    [key: string]: ChainBlockExplorer
  };
  contracts?: { [x: string]: ChainContract | { [sourceId: number]: ChainContract | undefined; } | undefined; ensRegistry?: ChainContract | undefined; ensUniversalResolver?: ChainContract | undefined; multicall3?: ChainContract | undefined; universalSignatureVerifier?: ChainContract | undefined; };
  custom?: Record<string, unknown>;
  fees?: ChainFees<undefined>;
  formatters?: undefined;
  id: 1337;
  name: "Seismic";
  nativeCurrency: {
    decimals: 18;
    name: "Ether";
    symbol: "ETH"
  };
  rpcUrls: {
    default: {
      http: readonly ["http://127.0.0.1:8545"];
      webSocket: readonly ["ws://127.0.0.1:8545"]
    }
  };
  serializers?: ChainSerializers<undefined, TransactionSerializable>;
  sourceId?: number;
  testnet?: boolean
}
```
